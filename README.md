```bash

PORT=<specified port no (e.g. 3000)>
AWS_ACCESS_KEY=<aws user access key>
AWS_SECRET_KEY=<aws user secret key>
AWS_REGION=<aws bucket & account region>
AWS_BUCKET=<aws bucket name>
```
